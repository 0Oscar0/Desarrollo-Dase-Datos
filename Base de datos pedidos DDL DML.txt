/*Crear las tablas

CREATE TABLE EMPLEADO(
EMPLEADOID int PRIMARY KEY NOT NULL,
NOMBRE char(30) NULL,
APELLIDO char(30) NULL,
FECHA_NAC date NULL,
REPORTA_A int NULL,
EXTENSION int NULL);

CREATE TABLE PROVEEDORE(
PROVEEDORID int PRIMARY KEY NOT NULL,
NOMBREPROV char(50) NOT NULL,
CONTACTO char(50) NOT NULL,
CELUPROV char(12) NULL,
FIJOPROV char(12) NULL
 );

CREATE TABLE CATEGORIA(
CATEGORIAID int PRIMARY KEY NOT NULL,
NOMBRECAT char(50) NOT NULL
 ) ;

CREATE TABLE CLIENTE(
CLIENTEID int PRIMARY KEY NOT NULL,
CEDULA_RUC char(10) NOT NULL,
NOMBRECIA char(30) NOT NULL,
NOMBRECONTACTO char(50) NOT NULL,
DIRECCIONCLI char(50) NOT NULL,
FAX char(12) NULL,
EMAIL char(50) NULL,
CELULAR char(12) NULL,
FIJO char(12) NULL
 );

CREATE TABLE PEDIDO(
PEDIDOID int PRIMARY KEY NOT NULL,
EMPLEADOID int NOT NULL,
CLIENTEID int NOT NULL,
FECHAORDEN date NOT NULL,
DESCUENTO int NULL
 );


CREATE TABLE DETALLE_PEDIDO(
PEDIDOID int NOT NULL,
DETALLEID int PRIMARY KEY NOT NULL,
PRODUCTOID int NOT NULL,
CANTIDAD int NOT NULL );

CREATE TABLE PRODUCTO(
PRODUCTOID int PRIMARY KEY NOT NULL,
PROVEEDORID int NOT NULL,
CATEGORIAID int NOT NULL,
DESCRIPCION char(50) NULL,
PRECIOUNIT numeric NOT NULL,
EXISTENCIA int NOT NULL
) ;
